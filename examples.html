function generateRandomArrays(val){
    var x,
    randArry = [];
    for(x = 0; x < val; x++){
        randArry.push(Math.floor((Math.random()*100) + 1))
    }
    return randArry;
}


var waves = Dlite.waveforms.domain(1, 100),
    defA = generateRandomArrays(50);


Dlite("waves")
    .setContent(waves(defA))
    .setSize(5)
    .setColor("#f00")
    .setAlignment(Dlite.align.HORIZONTAL)
    .showBackground(Dlite.background.SHOW)
    .show(null, 500, null, function(_){
            var arr = generateRandomArrays(50);
            _.setContent(waves(arr));
    });

function zeroPadNumber(number){
    var sNumber = String(number),
    PAD = "0",
    newNumber;
    if(typeof sNumber === "string"){
        if(sNumber.length === 1){
            newNumber = PAD.concat(sNumber);
            return newNumber;
        }
    }
    return sNumber;
}

var today = new Date();


Dlite("timer")
    .setContent(zeroPadNumber(today.getHours())+":"+zeroPadNumber(today.getMinutes())+":"+zeroPadNumber(today.getSeconds()))
    .setSize(6)
    .setColor("#00f")
    .setAlignment(Dlite.align.HORIZONTAL)
    .showBackground(Dlite.background.PARTIAL)
    .show(null, 1000, null,function(_){
        var s = new Date();
        _.setContent(zeroPadNumber(s.getHours())+":"+zeroPadNumber(s.getMinutes())+":"+zeroPadNumber(s.getSeconds()));
    });
